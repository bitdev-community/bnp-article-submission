import { MemoryRouter } from 'react-router-dom';
import { Link } from './link';

A navigation component that create a hyperlink using React Router DOM.
Use with `ThemeProvider` to enforce styling.

### Component usage

Use with a React Router DOM implementation to support routing

```js
<MemoryRouter>
  <Link to={'/'}>Hello world!</Link>
</MemoryRouter>
```

### Render hello world!

